class Result<T, E> {
  private constructor(private value: T | null, private error: E | null) {}

  static ok<T, E>(value: T): Result<T, E> {
    return new Result<T, E>(value, null);
  }

  static error<T, E>(error: E): Result<T, E> {
    return new Result<T, E>(null, error);
  }

  map<U>(fn: (value: T) => U): Result<U, E> {
    if (this.value !== null) {
      return Result.ok(fn(this.value));
    } else {
      return Result.error(this.error!);
    }
  }

  flatMap<U>(fn: (value: T) => Result<U, E>): Result<U, E> {
    if (this.value !== null) {
      return fn(this.value);
    } else {
      return Result.error(this.error!);
    }
  }

  isError(): boolean {
    return this.error !== null;
  }

  unwrap(): T {
    if (this.value !== null) {
      return this.value;
    } else {
      throw new Error("Tried to unwrap an error result");
    }
  }

  unwrapError(): E {
    if (this.error !== null) {
      return this.error;
    } else {
      throw new Error("Tried to unwrap an ok result");
    }
  }
}

// Ejemplo de uso
function divide(a: number, b: number): Result<number, string> {
  if (b === 0) {
    return Result.error("Divisi√≥n por cero");
  } else {
    return Result.ok(a / b);
  }
}
